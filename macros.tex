%% TODO: Use \trans in all parts of the lecture? We currently often
%% use \xrightarrow directly.
%%
%% We use two different macros here because we really want to have the
%% reduced size for \withcost (not least to fit things on the slides),
%% but we don't want to use this style for all transitions, not least
%% since as of this writing we often use \xrightarrow directly for
%% transitions in earlier parts of the lecture. I'm defining \trans
%% here as a synonym for \xrightarrow so that it is clear that in the
%% lecture chapters that mix \trans and \withcost, this is intentional
%% and not an accident. If we used \xrightarrow and \withcost, it
%% might look more like we meant to use \withcost everywhere.
\newcommand{\trans}[1]{\xrightarrow{#1}}
\newcommand{\withcost}[1]{\xrightarrow{\raisebox{-1.5pt}[0pt][0pt]{\tiny\ensuremath{#1}}}}
